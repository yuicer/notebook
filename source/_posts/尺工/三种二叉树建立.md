## 第一种
```bash
Node* CreatTree()				//无参数      
	{
		char ch;
		Node *T;
		cin >> ch;
		if(ch =='?')
		{
			T = NULL;
		}
		else
		{
			T = new Node;
			T->date = ch;
			T->lchlid = CreatTree();

			T->rchild = CreatTree();
		}
		return T;
	}
```
## 第二种
```bash
Node* CreatTree(Node*&T)			//有参数  注意会改变指针的值即T = NULL 所以必须用引用&
	{
		char ch;
		cin >> ch;
		if(ch =='?')
		{
			T = NULL;
		}
		else
		{
			T = new Node;
			T->date = ch;
			CreatTree(T->lchlid);

			CreatTree(T->rchild);
		}
		return T;
	}
```
