## 方法
```
window.onload = function () {
	var a = document.getElementById('div1')
	a.addEventListener('click', handle)
}

function handle(e) {
	//取参
	var val = document.getElementById('div0')
	//调用
	f(e, val)
}

function f(e, val) {
	console.log(e, val)
}
```
